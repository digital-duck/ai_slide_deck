1. Installation

# Install Prefect
pip install prefect

# Or with extras
pip install "prefect[dev,kubernetes]"

2. First Flow

from prefect import flow, task

@task
def say_hello(name: str) -> str:
    return f"Hello, {name}!"

@flow
def hello_world():
    greeting = say_hello("Prefect")
    print(greeting)

if __name__ == "__main__":
    hello_world()

3. Run It

python hello_flow.py
